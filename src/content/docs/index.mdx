## Send an email

```ts
import { Duta } from "@duta/sdk";

const duta = new Duta("duta_live_xxx");

await duta.emails.send({
  from: "hello@yourdomain.com",
  to: "user@example.com",
  subject: "Hello from Duta",
  html: "<p>It works!</p>",
});
```
